/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    vector<double> averageOfLevels(TreeNode* root) {
        queue<TreeNode *> q;
        if (root != nullptr)
            q.push(root);
        vector<double> res;

        while (!q.empty()){
            int len = q.size();
            auto t = len;
            double sum = 0.0;
            while (t --){
                auto tt = q.front();
                q.pop();
                sum += tt->val;
                if (tt->left != nullptr)
                    q.push(tt->left);
                if (tt->right != nullptr)
                    q.push(tt->right);
            }
            sum = sum / len;
            res.push_back(sum);
        }
        return res;
    }
};

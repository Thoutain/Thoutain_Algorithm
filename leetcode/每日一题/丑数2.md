class Solution {
public:
    int nthUglyNumber(int n) {
        if (n == 1)
            return 1;
        int count = 1;
        int now;
        int ugly = 2;
        while (count < n)
        {
            now = ugly ++;
            while (now % 2 == 0)
                now /= 2;
            while (now % 3 == 0)
                now /= 3;
            while (now % 5 == 0)
                now /= 5;
            if ( now == 0)
                count ++;
        }
        return (ugly - 1);
    }
};

class Solution {
public:
    int nthUglyNumber(int n) {
        vector<int> q(1, 1);
        int i = 0, j = 0, k = 0;
        while (-- n)
        {
            int t = min(q[i] * 2, min(q[j] * 3, q[k] * 5));
            q.push_back(t);
            if (q[i] * 2 == t)
                i ++;
            if (q[j] * 3 == t)
                j ++;
            if (q[k] * 5 == t)
                k ++;
        }
        return q.back();
    }
};

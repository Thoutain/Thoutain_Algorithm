### 设计一个支持 push ，pop ，top 操作，并能在常数时间内检索到最小元素的栈。

- push(x) ―― 将元素 x 推入栈中。
- pop()?―― 删除栈顶的元素。
- top()?―― 获取栈顶元素。
- getMin() ―― 检索栈中的最小元素。
```C++
//怎么不用额外空间  存差值

class MinStack {
    stack<int> stk;
    stack<int> mstk;
public:
    /** initialize your data structure here. */
    MinStack() {
        mstk.push(INT_MAX);
    }
    
    void push(int x) {
        stk.push(x);
        if (x < mstk.top())
            mstk.push(x);
        else
            mstk.push(mstk.top());
    }
    
    void pop() {
        stk.pop();
        mstk.pop();
    }
    
    int top() {
        return stk.top();
    }
    
    int getMin() {
        return mstk.top();
    }
};

/**
 * Your MinStack object will be instantiated and called as such:
 * MinStack* obj = new MinStack();
 * obj->push(x);
 * obj->pop();
 * int param_3 = obj->top();
 * int param_4 = obj->getMin();
 */
 ```
